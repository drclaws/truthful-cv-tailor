% Icons are visual only, but ActualText gives text extraction a readable label.
% Keep the actual contact/date/location values as visible text beside the icon.
\newcommand{\withactualtext}[2]{%
  \pdfliteral direct{/Span << /ActualText (#1) >> BDC}%
  #2%
  \pdfliteral direct{EMC}%
}
\newcommand{\cvicon}[2]{{%
  \color{CVBlue}%
  \withactualtext{#2}{\faIcon{#1}}%
}}
\newcommand{\cvmetaicon}[2]{{%
  \color{CVMuted}%
  \withactualtext{#2}{\faIcon{#1}}%
}}

% Contact text remains visible; icons only improve scan speed.
\newcommand{\contactitem}[3]{%
  \cvicon{#1}{#2: }\hspace{0.28em}{\bfseries #3}%
}

\newcommand{\contactlink}[4]{%
  \cvicon{#1}{#2: }\hspace{0.28em}\href{#3}{\bfseries #4}%
}

% Profile links show only the alias/handle visually, while text extraction gets
% the full generated URL. The href is also generated from the alias/handle.
\newcommand{\profilelink}[4]{%
  \cvicon{#1}{#2: }\hspace{0.28em}%
  \href{https://#3}{\bfseries\withactualtext{https://#3}{#4}}%
}

% Header tags are scan aids only; they must be backed by validated evidence.
\newcommand{\headertag}[1]{%
  \begingroup
  \setlength{\fboxsep}{0.22em}%
  \setlength{\fboxrule}{0.35pt}%
  \fcolorbox{CVRule}{white}{\footnotesize\bfseries\textcolor{CVText}{#1}}%
  \endgroup\hspace{0.4em}%
}

\newcommand{\cvsection}[1]{%
  \Needspace{3\baselineskip}%
  \vspace*{1.0em}%
  {\large\bfseries\color{CVText}\MakeUppercase{#1}}\par
  \vspace{-0.42em}{\color{CVText}\rule{\linewidth}{1.05pt}}\par
  \vspace{0.18em}%
}

\newcommand{\cvsubsection}[1]{%
  \vspace{0.16em}{\bfseries\color{CVBlue}#1}\par
}

  \vspace{-0.5em}{\color{CVRule}\rule{\linewidth}{0.35pt}}\par

% Skill tags remain real PDF text. Keep labels short enough to wrap compactly.
\newcommand{\skillchip}[1]{%
  \begingroup
  \setbox0=\hbox{\bfseries #1}%
  \leavevmode\hbox{%
    \rlap{\raisebox{-0.42em}[0pt][0pt]{%
      \textcolor{CVRule}{\rule{\wd0}{0.4pt}}%
    }}%
    \bfseries\textcolor{CVText}{#1}%
  }%
  \endgroup\hspace{0.72em}%
}

\newcommand{\skillgroup}[2]{%
  \vspace{0.24em}%
  {\bfseries\color{CVBlue}#1}\par
  \vspace{0.2em}%
  {\raggedright\setlength{\baselineskip}{1.4\baselineskip}#2\par}%
  \vspace{0.24em}{\color{CVRule}\rule{\linewidth}{0.35pt}}\par
}
